\question 
%
\begin{parts}
%
\part Show that the mapping cylinder of every map $f: \mathbb{S}^1 \rightarrow
\mathbb{S}^1$ is a CW complex.

\part Construct a $2$-dimensional CW complex that contains both an annulus 
$\mathbb{S} \times I$ and a M\"{o}bius band as deformation retracts.

\end{parts}
